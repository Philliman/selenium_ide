function delectAllItemsInCart
waitForVisible //span[@class='price']
storeText //span[@class='price'] myCartPrice
echo price = ${myCartPrice}
while javascript{storedVars['myCartPrice']!='0.00'}
label checkMyCartPrice
clickAndWait css=span.cart-total
clickAndWait link=Remove item
waitForVisible //span[@class='price']
storeText //span[@class='price'] myCartPrice
echo price = ${myCartPrice}
endWhile
endFunction
